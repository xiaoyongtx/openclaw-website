简单易用

几行代码即可启动你的 AI 助手

import { OpenClaw } from 'openclaw';

const bot = new OpenClaw({
  model: 'claude-sonnet',
  platforms: ['discord', 'telegram']
});

bot.onMessage('mention', async (ctx) => {
  const response = await ctx.think(ctx.message);
  ctx.reply(response);
});

bot.start();
// 🎉 你的 AI 助手已启动！
